## Семантика парсера:
// <- Комментарий.
[[Название Типа клетки]]
// Морфогены (M{i}) вырабатываются в определенном радиусе и взаимодействуют с текущей клеткой или её соседями.
// Активатор [] это список условий которые проверяются через "И".
// У каждого гена может быть несколько активаторов (перечисляются через запятую). Они проверяются через "ИЛИ".
M0: [M0] [!M1, !M2] {}

// Функции n(6) и type() проверяют тип клетки и количество её соседей.
// Радиус (задаётся в скобках, или значение по-умолчанию = 12).
M1(6): [n(6)] [Type(Название типа для проверки)] {}

// В отличае от активаторов [], деактиваторы {} блокируют производство морфогена, если хотя-бы один из них возвращает истину. 
// Таким образом сначала проверяются деактиваторы, если же их нет или они возвращают ложь, то проверяются активаторы.
M2(0): [] { n(4), n(5), n(6) }

// Таймер - это ген который запускается при любом активаторе и работает N тиков.
// Его можно применять в активаторах других генов.
T0(10): [M1] {}

// Условие которое определяет может ли клетка делится, по-умолчанию - ложно.
D: [] {}

// Цвет клетки в формате hex.
Color: #000000

// Условие дифференцировки клетки в другой тип.
Change(Название типа): [] {}
